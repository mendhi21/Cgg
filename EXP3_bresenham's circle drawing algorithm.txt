#include<stdio.h>
//#include<math.h>
#include<GL/gl.h>
#include<GL/glu.h>
#include<GL/glut.h>

//centre of the circle =320,240

int xc=320,yc=240;

//plot eight points using circle's symmetrical property
void plot_point(int x,int y)
	{
		glBegin(GL_POINTS);
		glVertex2i(x+xc,y+yc);
		glVertex2i(-x+xc,y+yc);
		glVertex2i(-x+xc,-y+yc);
		glVertex2i(x+xc,-y+yc);
		glVertex2i(y+xc,x+yc);
		glVertex2i(-y+xc,x+yc);
		glVertex2i(-y+xc,-x+yc);
		glVertex2i(y+xc,-x+yc);
		
		glEnd();
	}
	//function to draw a circle using bresenham's circle drawing algorithm
	
	void bresenham_circle(int r)
	{
		int x=0,y=r;
		float s=3-(2*r);
		
		plot_point(x,y);
		int k;
		while(x<=y)
		{
		if(s<=0)
		{
		s=s+(4*x)+6;
		x++;
		}
		else
		{
		s=s+(4*x)-(4*y)+10;
		x++;
		y=y-1;
		}
		plot_point(x,y);
		}
	glFlush();
	}
	
	//function to draw two concentric circles
	
	void con_cir(void)
	{
		glClear(GL_COLOR_BUFFER_BIT);
		
		int radius1=100,radius2=200,radius3=50;
		bresenham_circle(radius1);
		bresenham_circle(radius2);
		bresenham_circle(radius3);
	}
	
	void Init()
	{
		glClearColor(1,1,1,0);
		glColor3f(0,0,0);
		gluOrtho2D(0,640,0,480);
	}
	
	void main(int argc,char **argv)
	{
	glutInit(&argc,argv);
	glutInitDisplayMode(GLUT_SINGLE|GLUT_RGB);
	glutInitWindowPosition(0,0);
	glutInitWindowSize(640,480);
	glutCreateWindow("bresenham circle");
	Init();
	glutDisplayFunc(con_cir);
	glutMainLoop();
	}
